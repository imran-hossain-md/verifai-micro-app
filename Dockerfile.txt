# ১. অফিশিয়াল Deno ইমেজ ব্যবহার করা হচ্ছে
FROM denoland/deno:alpine-1.44.4

# ২. সার্ভারের ভেতরে কাজের জন্য একটি ডিরেক্টরি সেট করা
WORKDIR /app

# ৩. প্রজেক্টের সব ফাইল কন্টেইনারের ভেতরে কপি করা
COPY . .

# ৪. Deno ক্যাশ রান করা যাতে ডেপ্লয়মেন্ট দ্রুত হয়
RUN deno cache server.ts

# 📂 কন্টেইনারের ভেতরে পোর্ট এক্সপোজ করা
EXPOSE 3000

# 🏃‍♂️ কন্টেইনার চালু হওয়ার সময় সার্ভার রান করার কমান্ড
CMD ["run", "--allow-net", "--allow-read", "--allow-env", "server.ts"]
